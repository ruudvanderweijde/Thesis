\begin{prooftree}
    \AxiomC{$E \equiv \$ this \in C$}
    \UnaryInfC{$[E]$ <: object()} \noLine
    \UnaryInfC{$[E]$ = class(C) $\lor [E]$ :> class(C)}
\end{prooftree}
\begin{prooftree}
    \AxiomC{$E \equiv self \in C$}
    \UnaryInfC{$[E]$ <: object()} \noLine
    \UnaryInfC{$[self]$ = class(C)}
\end{prooftree}
\begin{prooftree}
    \AxiomC{$E \equiv parent \in C$}
    \UnaryInfC{$[E]$ <: object()} \noLine
    \UnaryInfC{$[E]$ :> class(C)}
\end{prooftree}
\begin{prooftree}
    \AxiomC{$E \equiv static \in C$}
    \UnaryInfC{$[E]$ <: object()} \noLine
    \UnaryInfC{( $[E]$ <: class(C) $\lor$ $[E]$ :> class(C) )}
\end{prooftree}
