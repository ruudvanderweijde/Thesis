\documentclass[border=10pt]{standalone}

\usepackage{graphicx}
\usepackage{forest}

\begin{document}

\begin{forest} for tree={
    edge path={\noexpand\path[\forestoption{edge}] (\forestOve{\forestove{@parent}}{name}.parent anchor) -- +(0,-12pt)-| (\forestove{name}.child anchor)\forestoption{edge label};}
}
[any(),name=any
    [array(any()),name=array]
    [boolean()]
    [float()
        [integer()]
    ]
    [object()
        [class(parent)
            [class(child1)
                [class(grandChild)]
            ]
            [class(child2)]
        ]
        [class(childless)]
    ]
    [resource()]
    [null()]
    [string()]
    [unset()]
]
\draw[->,dotted] (array) .. controls +(north west:2cm) and +(north:2cm).. (any);
\end{forest}

\end{document}
